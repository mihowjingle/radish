### If you really, really insist on local variables...

```
slot LocalVar some T { ; ugly name, TBD
    value: T
    
    constructors {
        new = (value: T) {
            value = value
        }
    }
    
    interface {
        set = (newValue) { value = newValue }
        get = () { value }
    }
}

;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;

root = () {
    number = LocalVar.new(5)
    write(number.get())
    number.set(7)
    write(number.get())
}

; will print 5 then 7
```
